------Steps and Commands to deploy jupyterhub------

#Create pvc using jupyterhub_pvc.yaml file
kubectl apply -f jupyterhub_pvc.yaml

#Create a copy_pod to copy config file to etc pvc || This step is required only initially
kubectl apply -f copy_pod.yaml
create a file in folder /etc/jupyterhub and name it as "jupyterhub_config.py"
copy the content of "jupyterhub_config.py"

#Delete the copy_pod 
kubectl delete -f copy_pod.yaml

#Apply the bd_jupyterhub.yaml file
kubectl apply -f bd_jupyterhub.yaml

#Verify pvc,statefulset and service
kubectl get pvc
kubectl get all | Select-String "jupyterhub"
